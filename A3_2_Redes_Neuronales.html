3. Procesamiento Mejorado

Primero har una mejora en la prediccin, utilizando Binarizacin adaptativa y limpieza de imagen, utilizando Binarizacin adaptativa en lugar de threshold fijo, Filtrado Gaussiano para reducir ruido y operaciones morfolgicas para limpiar la imagen

In[]:
def procesar_imagen_mejorada(ruta_imagen):
    # Cargar imagen
    imagen = cv2.imread(ruta_imagen, cv2.IMREAD_GRAYSCALE)

    # 1. Suavizado para reducir ruido
    imagen_suavizada = cv2.GaussianBlur(imagen, (3, 3), 0)

    # 2. Binarizacin adaptativa (MEJORA PRINCIPAL)
    imagen_binaria = cv2.adaptiveThreshold(imagen_suavizada, 255,
                                         cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
                                         cv2.THRESH_BINARY_INV, 11, 2)

    # 3. Operaciones morfolgicas para limpiar
    kernel = np.ones((2, 2), np.uint8)
    imagen_limpia = cv2.morphologyEx(imagen_binaria, cv2.MORPH_OPEN, kernel)

    # 4. Redimensionar
    imagen_redimensionada = cv2.resize(imagen_limpia, (28, 28), interpolation=cv2.INTER_AREA)

    # 5. Normalizar
    imagen_normalizada = imagen_redimensionada / 255.0

    return imagen, imagen_binaria, imagen_redimensionada, imagen_normalizada

Har otra mejora de prediccin, haciendo una validacin de confianza y anlisis de resultados, aadiendo Alertas para predicciones poco confiables y Retorna informacin completa para anlisis

In[]:
def predecir_digito_mejorado(imagen_normalizada, modelo, umbral_confianza=0.5):
    # Reformatear para el modelo
    imagen_entrada = imagen_normalizada.reshape(1, 28, 28, 1)

    # Realizar prediccin
    prediccion = modelo.predict(imagen_entrada, verbose=0)
    etiqueta_predicha = np.argmax(prediccion)
    confianza = np.max(prediccion)

    # Validar confianza
    if confianza < umbral_confianza:
        print(f"Advertencia: Confianza baja ({confianza:.3f})")

    return prediccion, etiqueta_predicha, confianza

def visualizar_procesamiento_completo(ruta_imagen, digito_real, modelo):
    # Procesar imagen con mtodo mejorado
    original, binaria, redimensionada, normalizada = procesar_imagen_mejorada(ruta_imagen)

    # Predecir
    prediccion, digito_predicho, confianza = predecir_digito_mejorado(normalizada, modelo)

    # Visualizacin completa
    fig, axes = plt.subplots(2, 3, figsize=(15, 8))

    # Fila 1: Procesamiento
    axes[0, 0].imshow(original, cmap='gray')
    axes[0, 0].set_title('Imagen Original')
    axes[0, 0].axis('off')

    axes[0, 1].imshow(binaria, cmap='gray')
    axes[0, 1].set_title('Binarizada Adaptativa')
    axes[0, 1].axis('off')

    axes[0, 2].imshow(redimensionada, cmap='gray')
    axes[0, 2].set_title('Redimensionada (28x28)')
    axes[0, 2].axis('off')

    # Fila 2: Anlisis
    axes[1, 0].imshow(normalizada, cmap='gray')
    axes[1, 0].set_title('Imagen Normalizada')
    axes[1, 0].axis('off')

    # Grfico de probabilidades
    axes[1, 1].bar(range(10), prediccion[0])
    axes[1, 1].set_title('Probabilidades por Dgito')
    axes[1, 1].set_xlabel('Dgito')
    axes[1, 1].set_ylabel('Probabilidad')
    axes[1, 1].set_xticks(range(10))

    # Resultados
    color = 'green' if digito_real == digito_predicho else 'red'
    axes[1, 2].text(0.1, 0.7, f'REAL: {digito_real}\nPRED: {digito_predicho}',
                   fontsize=14, fontweight='bold', color=color)
    axes[1, 2].text(0.1, 0.4, f'CONFIANZA: {confianza:.3f}', fontsize=12)
    axes[1, 2].axis('off')

    plt.tight_layout()
    plt.show()

    return digito_predicho, prediccion
